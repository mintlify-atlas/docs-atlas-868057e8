## Create Driver Profile

<api method="POST" path="/drivers">
  Creates a new driver profile with license information and optional vehicle.
</api>

### Body Parameters

<ParamField body="userId" type="string" required>
  UUID of the user account to associate with this driver profile
</ParamField>

<ParamField body="driverLicenseNumber" type="string" required>
  Driver license number (1-50 characters, must be unique)
</ParamField>

<ParamField body="driverLicenseExpirationDate" type="string" required>
  License expiration date in ISO 8601 format
</ParamField>

<ParamField body="driverLicensePictureUrl" type="string">
  URL to the driver license image
</ParamField>

<ParamField body="backgroundCheckStatus" type="enum">
  Background check status. Defaults to `pending_background_check`
  - `pending_background_check`
  - `approved`
  - `rejected`
</ParamField>

<ParamField body="backgroundCheckDate" type="string">
  Date when background check was performed (ISO 8601)
</ParamField>

<ParamField body="isApproved" type="boolean">
  Whether the driver is approved to operate. Defaults to `false`
</ParamField>

<ParamField body="emergencyContactInfo" type="object">
  Emergency contact information
  <Expandable title="properties">
    <ParamField body="name" type="string" required>
      Contact name (1-100 characters)
    </ParamField>
    <ParamField body="phoneNumber" type="string" required>
      Contact phone number (7-20 characters)
    </ParamField>
    <ParamField body="relationship" type="string" required>
      Relationship to driver (1-50 characters)
    </ParamField>
  </Expandable>
</ParamField>

<ParamField body="driverStatus" type="enum">
  Operational status of the driver. Defaults to `pending_docs`
  - `active` - Approved and actively driving
  - `suspended` - Temporarily suspended
  - `on_vacation` - On vacation
  - `pending_docs` - Awaiting documentation
  - `deactivated` - Account deactivated
</ParamField>

<ParamField body="paidPriorityUntil" type="string">
  Date until which the driver has paid priority status (ISO 8601)
</ParamField>

<ParamField body="initialVehicle" type="object">
  Optional vehicle to create alongside the driver profile during onboarding
</ParamField>

### Response

<ResponseField name="success" type="boolean">
  Indicates if the operation was successful
</ResponseField>

<ResponseField name="message" type="string">
  Response message
</ResponseField>

<ResponseField name="data" type="object">
  Created driver profile
  <Expandable title="properties">
    <ResponseField name="id" type="string">
      Driver profile UUID
    </ResponseField>
    <ResponseField name="userId" type="string">
      Associated user UUID
    </ResponseField>
    <ResponseField name="driverLicenseNumber" type="string">
      Driver license number
    </ResponseField>
    <ResponseField name="driverLicenseExpirationDate" type="string">
      License expiration date
    </ResponseField>
    <ResponseField name="backgroundCheckStatus" type="string">
      Background check status
    </ResponseField>
    <ResponseField name="isApproved" type="boolean">
      Approval status
    </ResponseField>
    <ResponseField name="driverStatus" type="string">
      Current driver operational status
    </ResponseField>
    <ResponseField name="createdAt" type="string">
      Creation timestamp
    </ResponseField>
    <ResponseField name="updatedAt" type="string">
      Last update timestamp
    </ResponseField>
  </Expandable>
</ResponseField>
